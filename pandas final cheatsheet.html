🚀 Flujo completo de limpieza — CSV / Excel workflow 
import pandas as pd

# ─── 1. CARGAR ────────────────────────────
df = pd.read_csv('ventas.csv', encoding='utf-8')
# df = pd.read_excel('ventas.xlsx', sheet_name='Datos')

# ─── 2. INSPECCIONAR ──────────────────────
print(df.shape)
df.info()
print(df.isnull().sum())

# ─── 3. LIMPIAR NOMBRES DE COLUMNAS ───────
df.columns = (df.columns
              .str.strip().str.lower()
              .str.replace(' ', '_')
              .str.replace(r'[^\w]', '', regex=True))

# ─── 4. ELIMINAR DUPLICADOS ───────────────
df.drop_duplicates(subset=['id'], keep='first', inplace=True)

# ─── 5. TRATAR NULOS ──────────────────────
df.dropna(subset=['id', 'fecha'], inplace=True)  # obligatorias
df['monto'].fillna(0, inplace=True)               # numéricas → 0
df['categoria'].fillna('Sin categoría', inplace=True)

# ─── 6. LIMPIAR TEXTO ─────────────────────
str_cols = df.select_dtypes(include='object').columns
df[str_cols] = df[str_cols].apply(lambda c: c.str.strip())

# ─── 7. CONVERTIR TIPOS ───────────────────
df['fecha'] = pd.to_datetime(df['fecha'], errors='coerce')
df['monto'] = pd.to_numeric(df['monto'], errors='coerce')

# ─── 8. VALIDAR ───────────────────────────
assert df.duplicated().sum() == 0, '¡Hay duplicados!'
assert df['monto'].min() >= 0, 'Hay montos negativos'

# ─── 9. EXPORTAR ──────────────────────────
df.to_csv('ventas_limpio.csv', index=False, encoding='utf-8')
df.to_excel('ventas_limpio.xlsx', index=False)

print(f'✅ Limpieza completa. Filas: {len(df)}')

Lambda Functions — Funciones en una línea

💡
Una lambda es una función anónima de una sola línea. Se usa principalmente con .apply() para transformar columnas y con .agg() para agregaciones personalizadas. Sintaxis: lambda x: operación
apply() — transformar columna básico 
# Convertir a dólares
df['salario_usd'] = df['salario'].apply(lambda x: x / 20)

# Aumentar 10%
df['sal_aum'] = df['salario'].apply(lambda x: x * 1.10)

# Clasificar: 2 opciones (if / else)
df['cat'] = df['edad'].apply(
    lambda x: 'Joven' if x < 30 else 'Adulto'
)

# Clasificar: 3 opciones (if / elif / else)
df['nivel'] = df['edad'].apply(
    lambda x: 'Junior' if x < 30
    else ('Senior' if x < 45 else 'Expert')
)
agg() — agregaciones personalizadas intermedio 
# Diferencia entre máximo y mínimo por grupo
df.groupby('departamento')['salario'].agg(
    lambda x: x.max() - x.min()
)

# Promedio solo de valores mayores a 25
df.groupby('departamento')['edad'].agg(
    lambda x: x[x > 25].mean()
)

# Suma de valores mayores al umbral
df.groupby('departamento')['salario'].agg(
    lambda x: x[x > 2000].sum()
)
Plantillas de lambda listas para copiar plantillas 
# PLANTILLA 1: Operación matemática
df['nueva'] = df['col'].apply(lambda x: x * NÚMERO)

# PLANTILLA 2: if/else (2 categorías)
df['nueva'] = df['col'].apply(
    lambda x: 'TEXTO1' if x > NÚMERO else 'TEXTO2'
)

# PLANTILLA 3: if/elif/else (3 categorías)
df['nueva'] = df['col'].apply(
    lambda x: 'CAT1' if x < NUM1
    else ('CAT2' if x <= NUM2 else 'CAT3')
)

# PLANTILLA 4: Diferencia max-min por grupo
df.groupby('grupo')['col'].agg(lambda x: x.max() - x.min())

# PLANTILLA 5: Filtrar y calcular por grupo
df.groupby('grupo')['col'].agg(lambda x: x[x > NÚMERO].mean())

Axis — Dirección de las Operaciones

🧭
axis=0 opera verticalmente (a lo largo de las filas → resultado por columna). axis=1 opera horizontalmente (a lo largo de las columnas → resultado por fila). Regla mnemotécnica: "0" parece una columna vertical, "1" parece una línea horizontal.
sum() y mean() con axis intermedio 
df = pd.DataFrame({
    'A': [10, 20, 30],
    'B': [15, 25, 35],
    'C': [20, 30, 40]
})

# axis=0 → suma cada COLUMNA (↓ vertical)
print(df.sum(axis=0))
# A: 60  B: 75  C: 90

# axis=1 → suma cada FILA (→ horizontal)
print(df.sum(axis=1))
# fila 0: 45  fila 1: 75  fila 2: 105

# Promedio por columna
print(df.mean(axis=0))  # A:20  B:25  C:30

# Promedio por fila
print(df.mean(axis=1))  # fila0:15  fila1:25  fila2:35
drop() y apply() con axis intermedio 
# drop() — axis=0 elimina FILAS, axis=1 elimina COLUMNAS
df.drop(0, axis=0)       # elimina la fila 0
df.drop('A', axis=1)     # elimina la columna A
df.drop([0, 5], axis=0)  # elimina filas 0 y 5

# apply() — axis=0 aplica a cada COLUMNA
df.apply(lambda x: x.max(), axis=0)
# → máximo de cada columna

# apply() — axis=1 aplica a cada FILA
df.apply(lambda x: x.max(), axis=1)
# → máximo de cada fila
Casos de uso reales casos 
# Sumar todas las ventas de cada producto (por columna)

🗺️ Mapa de problemas típicos en datasets reales ref esencial
💰 Columnas de dinero/número
  • $1,234.56 → símbolo $ y comas

  • 15% → símbolo %
  • 1.234,56 → formato europeo

📝 Columnas de texto
  • \xa0 → espacio invisible (nbsp)
  • " ana " → espacios extra
  • "VENTAS" / "ventas" → mayúsculas mixtas
  • "Ana\n" → salto de línea oculto
  • → mismo valor, múltiples formas
∅ Valores nulos / vacíos
  • NaN → nulo numérico de pandas
  • None → nulo de Python
  • " " → string vacío o solo espacios
  • "N/A" "–" "null" "na" → nulos disfrazados
  • → isna() no los detecta todos
📅 Tipos de dato incorrectos
  • object en lugar de float64
  • object en lugar de datetime64
  • float64 en lugar de int64
  • int64 en lugar de category
  • → operaciones matemáticas fallan
📋 Columnas y filas
  • "Nombre Completo " → espacios en header
  • "Unnamed: 0" → índice guardado como col
  • Filas totalmente duplicadas
  • Filas con casi todo NaN

  • Columna numérica con dtype object

import pandas as pd

def diagnosticar(df):
    print("=== FORMA ===")
    print(df.shape)

    print("\n=== TIPOS DE DATO ===")
    print(df.dtypes)

    print("\n=== NULOS ===")
    nulos = df.isna().sum()
    print(nulos[nulos > 0])

    print("\n=== DUPLICADOS ===")
    print(f"Filas duplicadas: {df.duplicated().sum()}")

    print("\n=== CARACTERES RAROS EN COLUMNAS ===")
    for col in df.columns:
        if '\xa0' in repr(col) or col != col.strip():
            print(f"  ⚠️  {repr(col)}")

    print("\n=== STRINGS VACÍOS ===")
    for col in df.select_dtypes(include='object').columns:
        vacios = (df[col] == '').sum()
        if vacios > 0:
            print(f"  {col}: {vacios} strings vacíos")

    for col in df.select_dtypes(include='object').columns:
        convertido = pd.to_numeric(df[col], errors='coerce')
        pct_ok = convertido.notna().mean()
        if pct_ok > 0.7:  # más del 70% parece número
