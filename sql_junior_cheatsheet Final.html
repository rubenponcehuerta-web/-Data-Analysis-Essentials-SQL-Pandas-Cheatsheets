SELECT — Consultas Básicas

💡
El SELECT es la base de todo. Define qué columnas quieres ver. La cláusula FROM indica de qué tabla. Orden obligatorio: SELECT → FROM → WHERE → GROUP BY → HAVING → ORDER BY → LIMIT.
Selección básicabásico
-- Todas las columnas
SELECT *
FROM empleados;

-- Columnas específicas
SELECT nombre, apellido, salario
FROM empleados;

-- Con alias de columna
SELECT
    nombre        AS Nombre,
    salario       AS "Salario Mensual",
    departamento  AS Depto
FROM empleados;

-- Alias de tabla
SELECT e.nombre, e.salario
FROM empleados AS e;
DISTINCT y LIMITbásico
-- Valores únicos (sin repetidos)
SELECT DISTINCT departamento
FROM empleados;

-- Distinct en varias columnas
SELECT DISTINCT ciudad, pais
FROM clientes;

-- Limitar resultados
SELECT *
FROM productos
LIMIT 10;

-- LIMIT + OFFSET (paginación)
SELECT *
FROM productos
LIMIT 10 OFFSET 20;  -- página 3
Operaciones en SELECTbásico
-- Aritmética
SELECT
    nombre,
    salario,
    salario * 1.1          AS salario_con_aumento,
    salario * 12           AS salario_anual,
    precio - descuento     AS precio_final,
    total / cantidad       AS precio_unitario
FROM empleados;

-- Concatenar texto directo
SELECT
    nombre || ' ' || apellido AS nombre_completo
FROM empleados;
-- En MySQL: CONCAT(nombre,' ',apellido)

Filtros — WHERE

Operadores de comparaciónbásico
SELECT * FROM empleados
WHERE salario > 30000;      -- mayor que

WHERE salario >= 30000;     -- mayor o igual
WHERE salario < 50000;      -- menor que
WHERE salario <= 50000;     -- menor o igual
WHERE salario = 40000;      -- igual
WHERE salario <> 40000;     -- distinto (también !=)
WHERE nombre = 'Ana';       -- texto (comillas simples)
AND · OR · NOTbásico
-- AND: ambas condiciones
SELECT * FROM empleados
WHERE departamento = 'Ventas'
  AND salario > 35000;

-- OR: al menos una
SELECT * FROM empleados
WHERE ciudad = 'CDMX'
   OR ciudad = 'MTY';

-- NOT: negar condición
SELECT * FROM empleados
WHERE NOT departamento = 'RRHH';

-- Combinar con paréntesis (muy importante)
WHERE (ciudad = 'CDMX' OR ciudad = 'MTY')
  AND salario > 40000;
BETWEEN · IN · LIKEbásico
-- BETWEEN: rango (inclusivo en ambos extremos)
SELECT * FROM empleados
WHERE salario BETWEEN 30000 AND 60000;

WHERE fecha_ingreso BETWEEN '2022-01-01'
                        AND '2023-12-31';

-- IN: lista de valores
SELECT * FROM empleados
WHERE departamento IN ('Ventas', 'Finanzas', 'TI');

-- NOT IN
WHERE ciudad NOT IN ('GDL', 'MTY');

-- LIKE: patrón de texto
WHERE nombre LIKE 'A%'      -- empieza con A
WHERE nombre LIKE '%ez'     -- termina en ez
WHERE email  LIKE '%@gmail%'-- contiene @gmail
WHERE codigo LIKE 'A_3'     -- _ = un carácter
IS NULL · IS NOT NULLbásico
-- Buscar nulos (NUNCA usar = NULL)
SELECT * FROM clientes
WHERE telefono IS NULL;

-- Sin nulos
SELECT * FROM clientes
WHERE email IS NOT NULL;

-- Combinar con otros filtros
SELECT nombre, email, telefono
FROM clientes
WHERE email IS NOT NULL
  AND activo = 1;

ORDER BY — Ordenar Resultados

Ordenamiento básicobásico
-- Ascendente (default)
SELECT nombre, salario
FROM empleados
ORDER BY salario;

-- Descendente
SELECT nombre, salario
FROM empleados
ORDER BY salario DESC;

-- Múltiples columnas
SELECT departamento, nombre, salario
FROM empleados
ORDER BY departamento ASC, salario DESC;
-- → primero ordena por depto,
--   dentro del mismo depto por salario desc
ORDER BY avanzadointermedio
-- Por posición de columna
SELECT nombre, apellido, salario
FROM empleados
ORDER BY 3 DESC;  -- 3ra columna = salario

-- Por alias
SELECT nombre, salario * 12 AS anual
FROM empleados
ORDER BY anual DESC;

-- Nulos al final / al inicio
ORDER BY fecha DESC NULLS LAST;
ORDER BY fecha ASC  NULLS FIRST;

-- Con LIMIT: top 5 mejor pagados
SELECT nombre, salario
FROM empleados
ORDER BY salario DESC
LIMIT 5;

Funciones de Agregación

✅
Las funciones agregadas calculan un único valor a partir de múltiples filas. Siempre ignoran NULLs (excepto COUNT(*)). Se usan junto a GROUP BY cuando quieres resultados por grupos.
COUNTbásico
-- Contar total de filas
SELECT COUNT(*) AS total
FROM empleados;

-- Contar filas no nulas en una columna
SELECT COUNT(email) AS con_email
FROM clientes;

-- Contar valores únicos
SELECT COUNT(DISTINCT departamento) AS num_deptos
FROM empleados;

-- COUNT con filtro
SELECT COUNT(*) AS emp_ventas
FROM empleados
WHERE departamento = 'Ventas';
SUM · AVGbásico
-- Suma total
SELECT SUM(ventas) AS total_ventas
FROM ordenes;

-- Suma condicional
SELECT SUM(monto) AS ventas_2024
FROM ordenes
WHERE YEAR(fecha) = 2024;

-- Promedio
SELECT AVG(salario) AS salario_promedio
FROM empleados;

-- Promedio con redondeo
SELECT ROUND(AVG(salario), 2) AS promedio
FROM empleados;

-- SUM y AVG juntos
SELECT
    SUM(monto)          AS total,
    AVG(monto)          AS promedio,
    COUNT(*)           AS num_ordenes
FROM ordenes;
MIN · MAXbásico
-- Mínimo y máximo numérico
SELECT
    MIN(salario) AS salario_min,
    MAX(salario) AS salario_max
FROM empleados;

-- También funciona con fechas
SELECT
    MIN(fecha_pedido) AS primer_pedido,
    MAX(fecha_pedido) AS ultimo_pedido
FROM ordenes;

-- Y con texto (orden alfabético)
SELECT
    MIN(nombre) AS primero_alfa,
    MAX(nombre) AS ultimo_alfa
FROM clientes;

GROUP BY & HAVING

⚠️
Regla de oro: Todo lo que esté en el SELECT y no sea una función agregada, debe estar en el GROUP BY. HAVING filtra grupos (como WHERE pero después de agrupar).
GROUP BY básicointermedio
-- Total de ventas por departamento
SELECT
    departamento,
    COUNT(*)       AS num_empleados,
    SUM(salario)    AS gasto_total,
    AVG(salario)    AS salario_prom,
    MIN(salario)    AS salario_min,
    MAX(salario)    AS salario_max
FROM empleados
GROUP BY departamento
ORDER BY gasto_total DESC;

-- Agrupar por múltiples columnas
SELECT
    departamento,
    ciudad,
    COUNT(*) AS empleados
FROM empleados
GROUP BY departamento, ciudad
ORDER BY departamento, empleados DESC;
HAVING — filtrar gruposintermedio
-- Departamentos con más de 5 empleados
SELECT
    departamento,
    COUNT(*) AS num_empleados
FROM empleados
GROUP BY departamento
HAVING COUNT(*) > 5;

-- WHERE + GROUP BY + HAVING juntos
SELECT
    departamento,
    AVG(salario) AS prom
FROM empleados
WHERE activo = 1             -- 1° filtra filas
GROUP BY departamento         -- 2° agrupa
HAVING AVG(salario) > 40000  -- 3° filtra grupos
ORDER BY prom DESC;           -- 4° ordena

-- HAVING con alias (algunos dialectos)
HAVING prom > 40000  -- PostgreSQL / MySQL 8+

JOINs — Unir Tablas

🔗
Los JOINs combinan filas de dos o más tablas basándose en una columna relacionada (generalmente una clave foránea). Siempre usa alias de tabla para que las queries sean legibles.
INNER JOIN — solo coincidenciasintermedio
-- Solo filas que tienen match en AMBAS tablas
SELECT
    e.nombre,
    e.salario,
    d.nombre_depto
FROM empleados      AS e
INNER JOIN departamentos AS d
    ON e.depto_id = d.id;

-- Con filtro adicional
SELECT
    e.nombre,
    d.nombre_depto
FROM empleados      AS e
INNER JOIN departamentos AS d
    ON e.depto_id = d.id
WHERE e.salario > 40000
ORDER BY e.nombre;
LEFT JOIN — todos del izquierdointermedio
-- TODAS las filas del lado izquierdo
-- + coincidencias del derecho (NULL si no hay)
SELECT
    c.nombre          AS cliente,
    o.id              AS orden_id,
    o.monto
FROM clientes AS c
LEFT JOIN ordenes AS o
    ON c.id = o.cliente_id;
-- Clientes SIN ordenes aparecen con NULL

-- Truco: encontrar clientes sin órdenes
SELECT c.nombre
FROM clientes AS c
LEFT JOIN ordenes AS o
    ON c.id = o.cliente_id
WHERE o.id IS NULL;
RIGHT JOIN · FULL OUTER JOINintermedio
-- RIGHT JOIN: todos del lado derecho
SELECT
    e.nombre,
    d.nombre_depto
FROM empleados         AS e
RIGHT JOIN departamentos AS d
    ON e.depto_id = d.id;
-- Deptos sin empleados aparecen con NULL

-- FULL OUTER JOIN: todos de ambos lados
-- (no disponible en MySQL, usar UNION)
SELECT e.nombre, d.nombre_depto
FROM empleados AS e
FULL OUTER JOIN departamentos AS d
    ON e.depto_id = d.id;
-- Incluye: matches + sin match de ambos lados
JOIN doble — 3 tablasintermedio
-- Empleados con su depto Y su manager
SELECT
    e.nombre          AS empleado,
    d.nombre_depto    AS departamento,
    m.nombre          AS manager
FROM      empleados     AS e
INNER JOIN departamentos AS d
    ON e.depto_id = d.id
LEFT JOIN  empleados     AS m
    ON e.manager_id = m.id
ORDER BY d.nombre_depto, e.nombre;
JOIN triple — 4 tablasintermedio
-- Órdenes con cliente, producto y vendedor
SELECT
    o.id              AS orden,
    c.nombre          AS cliente,
    p.nombre          AS producto,
    v.nombre          AS vendedor,
    o.monto
FROM       ordenes   AS o
INNER JOIN clientes  AS c
    ON o.cliente_id = c.id
INNER JOIN productos AS p
    ON o.producto_id = p.id
INNER JOIN empleados AS v
    ON o.vendedor_id = v.id
WHERE o.fecha >= '2024-01-01'
ORDER BY o.monto DESC;
JOIN cuádruple — 5 tablasintermedio
-- Reporte completo de envíos
SELECT
    o.id             AS orden,
    c.nombre         AS cliente,
    p.nombre         AS producto,
    cat.nombre       AS categoria,
    env.estado       AS envio,
    o.fecha,
    o.monto
FROM       ordenes    AS o
INNER JOIN clientes   AS c
    ON o.cliente_id = c.id
INNER JOIN productos  AS p
    ON o.producto_id = p.id
INNER JOIN categorias AS cat
    ON p.categoria_id = cat.id
LEFT JOIN  envios     AS env
    ON o.id = env.orden_id
WHERE YEAR(o.fecha) = 2024
ORDER BY o.fecha DESC;
JOIN + GROUP BY combinadosintermedio
-- Ventas totales por cliente
SELECT
    c.nombre,
    COUNT(o.id)     AS num_ordenes,
    SUM(o.monto)    AS total_gastado,
    AVG(o.monto)    AS ticket_promedio,
    MAX(o.fecha)    AS ultima_compra
FROM clientes  AS c
LEFT JOIN ordenes AS o
    ON c.id = o.cliente_id
GROUP BY c.id, c.nombre
HAVING SUM(o.monto) > 1000
ORDER BY total_gastado DESC;
UNION & UNION ALLintermedio
-- UNION: combina resultados, elimina duplicados
SELECT nombre, email FROM clientes
UNION
SELECT nombre, email FROM proveedores;

-- UNION ALL: incluye duplicados (más rápido)
SELECT nombre, 'cliente' AS tipo FROM clientes
UNION ALL
SELECT nombre, 'proveedor'        FROM proveedores;

-- Las columnas deben coincidir en número y tipo
-- El nombre de columna lo da la 1ra query

Subconsultas (Subqueries)

🪆
Una subconsulta es un SELECT dentro de otro SELECT. Va entre paréntesis. Se puede usar en el WHERE, en el FROM (tabla derivada) o en el SELECT (subconsulta escalar).
¿Cuándo usar Subqueries? guía de uso
✅ Úsalas cuando...
  • → Necesitas filtrar con el resultado de otra query (ej: salario mayor al promedio)
  • → Quieres saber si existe o no existe algo en otra tabla (EXISTS)
  • → Usas IN / NOT IN con una lista dinámica generada por otra query
  • → La lógica solo se usa una vez y es sencilla de entender inline
  • → Necesitas un valor escalar único dentro del SELECT (ej: mostrar el máximo global junto a cada fila)
❌ Evítalas cuando...
  • → La lógica se repite en varios lugares → usa CTE
  • → La subconsulta está en el SELECT y se ejecuta una vez por fila → muy lento, usar Window Function
  • → Tienes 3+ subqueries anidadas → el código se vuelve ilegible, prefiere CTEs
  • → Necesitas comparar filas entre sí (ej: vs. fila anterior) → usa Window Function
📌 Casos de uso reales
  • → Clientes que nunca compraron (NOT IN / NOT EXISTS)
  • → Productos con precio mayor al promedio de su categoría
  • → Empleados del departamento con más ventas
  • → Filtrar por una métrica calculada (WHERE monto > (SELECT AVG...))
Subconsulta en WHEREintermedio
-- Empleados con salario mayor al promedio
SELECT nombre, salario
FROM empleados
WHERE salario > (
    SELECT AVG(salario)
    FROM empleados
);

-- Con IN: clientes que compraron en 2024
SELECT nombre
FROM clientes
WHERE id IN (
    SELECT DISTINCT cliente_id
    FROM ordenes
    WHERE YEAR(fecha) = 2024
);

-- Con NOT IN: clientes que NUNCA compraron
SELECT nombre
FROM clientes
WHERE id NOT IN (
    SELECT DISTINCT cliente_id
    FROM ordenes
    WHERE cliente_id IS NOT NULL
);
Subconsulta en FROM (tabla derivada)intermedio
-- Promedio de ventas por depto, luego filtrar
SELECT depto, promedio
FROM (
    SELECT
        departamento    AS depto,
        AVG(salario)  AS promedio
    FROM empleados
    GROUP BY departamento
) AS resumen            -- alias obligatorio
WHERE promedio > 45000
ORDER BY promedio DESC;

-- Top producto por categoría
SELECT cat, nombre_prod, max_precio
FROM (
    SELECT
        categoria AS cat,
        nombre    AS nombre_prod,
        MAX(precio) AS max_precio
    FROM productos
    GROUP BY categoria, nombre
) AS sub
ORDER BY cat, max_precio DESC;
Subconsulta escalar en SELECTintermedio
-- Mostrar salario vs. promedio global
SELECT
    nombre,
    salario,
    (SELECT AVG(salario) FROM empleados) AS prom_global,
    salario - (SELECT AVG(salario) FROM empleados) AS diferencia
FROM empleados
ORDER BY diferencia DESC;
-- ⚠ Cuidado: se ejecuta una vez por fila → lento
--   Mejor usar CTE o Window Function
EXISTS / NOT EXISTSintermedio
-- EXISTS: ¿existe al menos una fila?
SELECT c.nombre
FROM clientes AS c
WHERE EXISTS (
    SELECT 1
    FROM ordenes AS o
    WHERE o.cliente_id = c.id
      AND o.monto > 5000
);

-- NOT EXISTS: clientes sin ninguna orden
SELECT c.nombre
FROM clientes AS c
WHERE NOT EXISTS (
    SELECT 1
    FROM ordenes AS o
    WHERE o.cliente_id = c.id
);
-- EXISTS es más eficiente que NOT IN
-- cuando hay NULLs en la subconsulta

CTEs — Common Table Expressions

✅
Un CTE es como una tabla temporal con nombre que existe solo para esa query. Hace el código más legible que las subconsultas anidadas. Se define con WITH nombre AS (...) antes del SELECT principal.
¿Cuándo usar CTEs? guía de uso
✅ Úsalos cuando...
  • → Necesitas reutilizar la misma lógica varias veces en la misma query
  • → Tienes subqueries anidadas que hacen el código ilegible
  • → Quieres dividir un problema complejo en pasos lógicos con nombre
  • → Necesitas hacer un JOIN sobre un resultado agregado (ej: JOIN contra un GROUP BY)
  • → Quieres que tu query se lea como una historia paso a paso
❌ Evítalos cuando...
  • → La lógica es tan simple que una subquery inline es más clara
  • → Necesitas comparar filas entre sí o calcular rankings → usa Window Function
  • → El CTE se usaría solo una vez y es de una sola línea (sobrecomplica sin beneficio)
📌 Casos de uso reales
  • → Comparar ventas de este año vs. año anterior (dos CTEs separados)
  • → Calcular el promedio por depto y luego JOIN para comparar cada empleado
  • → Construir un reporte en pasos: clientes → sus órdenes → totales → ranking
  • → Reemplazar una subquery en el FROM que se repite 3+ veces
💡 CTE vs Subquery: funcionalmente son equivalentes en la mayoría de casos. La diferencia es legibilidad. Si tienes que explicarlo a alguien más → usa CTE. Si es algo rápido de una línea → subquery está bien.
CTE básicointermedio
-- Misma lógica que subquery, más legible
WITH empleados_activos AS (
    SELECT id, nombre, salario, depto_id
    FROM empleados
    WHERE activo = 1
)
SELECT
    nombre,
    salario
FROM empleados_activos
WHERE salario > 40000
ORDER BY salario DESC;

-- También puedes hacer JOIN contra el CTE
WITH activos AS (
    SELECT * FROM empleados WHERE activo = 1
)
SELECT a.nombre, d.nombre_depto
FROM activos AS a
INNER JOIN departamentos AS d
    ON a.depto_id = d.id;
Múltiples CTEsintermedio
-- Varios CTEs separados por coma
WITH
ventas_2024 AS (
    SELECT cliente_id, SUM(monto) AS total
    FROM ordenes
    WHERE YEAR(fecha) = 2024
    GROUP BY cliente_id
),
ventas_2023 AS (
    SELECT cliente_id, SUM(monto) AS total
    FROM ordenes
    WHERE YEAR(fecha) = 2023
    GROUP BY cliente_id
)
SELECT
    c.nombre,
    v24.total  AS ventas_24,
    v23.total  AS ventas_23,
    v24.total - v23.total AS diferencia
FROM clientes   AS c
LEFT JOIN ventas_2024 AS v24 ON c.id = v24.cliente_id
LEFT JOIN ventas_2023 AS v23 ON c.id = v23.cliente_id
ORDER BY diferencia DESC;
CTE para reutilizar lógica complejaintermedio
-- Reporte: empleados sobre el prom de su depto
WITH prom_por_depto AS (
    SELECT
        depto_id,
        AVG(salario) AS prom_depto
    FROM empleados
    GROUP BY depto_id
)
SELECT
    e.nombre,
    e.salario,
    p.prom_depto,
    e.salario - p.prom_depto AS diferencia
FROM empleados        AS e
INNER JOIN prom_por_depto AS p
    ON e.depto_id = p.depto_id
WHERE e.salario > p.prom_depto
ORDER BY diferencia DESC;

Window Functions — Funciones de Ventana

⚠️
A diferencia de GROUP BY, las window functions no colapsan filas. Calculan un valor para cada fila basándose en un grupo relacionado. Usan la cláusula OVER(). Son lo más poderoso del SQL intermedio.
¿Cuándo usar Window Functions? guía de uso
✅ Úsalas cuando...
  • → Quieres rankear filas (top 3 por grupo, el mejor vendedor por región)
  • → Necesitas comparar una fila con la anterior o siguiente (LAG / LEAD)
  • → Quieres un acumulado progresivo (suma corrida, conteo acumulado)
  • → Necesitas agregar sin perder el detalle de las filas (ej: mostrar salario + promedio del depto en la misma fila)
  • → Quieres calcular % del total manteniendo el detalle por fila
  • → Necesitas el primer o último valor dentro de un grupo sin GROUP BY
❌ Evítalas cuando...
  • → Solo necesitas un resumen/total → GROUP BY + agregación es más simple
  • → Solo quieres filtrar basándote en otra tabla → usa subquery o JOIN
  • → Tu motor de BD no las soporta (MySQL < 8.0, SQLite antiguo)
📌 Casos de uso reales
  • → Top 1 empleado por departamento (ROW_NUMBER + CTE)
  • → Crecimiento % mes a mes en ventas (LAG)
  • → Suma acumulada de ingresos por fecha
  • → Mostrar cada venta con el total del vendedor en la misma fila
  • → Dividir clientes en cuartiles de gasto (NTILE)
💡 Window vs GROUP BY: GROUP BY colapsa las filas en una por grupo. Las Window Functions mantienen todas las filas y agregan una columna calculada. Si necesitas el detalle Y el resumen en la misma fila → Window Function.
ROW_NUMBER · RANK · DENSE_RANKintermedio
-- ROW_NUMBER: número único por fila
SELECT
    nombre,
    salario,
    ROW_NUMBER() OVER (ORDER BY salario DESC) AS fila
FROM empleados;

-- RANK: con saltos (dos en 1° → el siguiente es 3°)
SELECT
    nombre, salario,
    RANK() OVER (ORDER BY salario DESC) AS rank
FROM empleados;

-- DENSE_RANK: sin saltos (dos en 1° → el siguiente es 2°)
SELECT
    nombre, salario,
    DENSE_RANK() OVER (ORDER BY salario DESC) AS drank
FROM empleados;

-- Ranking DENTRO de cada departamento
SELECT
    nombre,
    departamento,
    salario,
    RANK() OVER (
        PARTITION BY departamento
        ORDER BY salario DESC
    ) AS rank_depto
FROM empleados;
SUM · AVG · COUNT con OVERintermedio
-- Total global sin GROUP BY (mantiene todas las filas)
SELECT
    nombre,
    salario,
    SUM(salario) OVER() AS suma_global,
    AVG(salario) OVER() AS prom_global
FROM empleados;

-- Agrupado por ventana (PARTITION BY)
SELECT
    nombre,
    departamento,
    salario,
    AVG(salario) OVER (
        PARTITION BY departamento
    ) AS prom_depto,
    salario - AVG(salario) OVER (
        PARTITION BY departamento
    ) AS vs_promedio
FROM empleados;

-- % del total por grupo
SELECT
    nombre,
    monto,
    ROUND(monto * 100.0 /
        SUM(monto) OVER(), 2) AS pct_total
FROM ventas;
SUM acumulada (Running Total)intermedio
-- Suma acumulada por fecha
SELECT
    fecha,
    monto,
    SUM(monto) OVER (
        ORDER BY fecha
    ) AS acumulado
FROM ventas;

-- Acumulado DENTRO de cada año
SELECT
    fecha,
    monto,
    YEAR(fecha) AS anio,
    SUM(monto) OVER (
        PARTITION BY YEAR(fecha)
        ORDER BY fecha
    ) AS acumulado_anio
FROM ventas;

-- Conteo acumulado
SELECT
    fecha,
    COUNT(*) OVER (ORDER BY fecha) AS total_acum
FROM ordenes;
LAG · LEAD — filas anterior/siguienteintermedio
-- LAG: valor de la fila ANTERIOR
SELECT
    fecha,
    monto,
    LAG(monto) OVER (ORDER BY fecha) AS monto_anterior,
    monto - LAG(monto) OVER (ORDER BY fecha) AS diferencia
FROM ventas;

-- LAG con N filas atrás y valor default
LAG(monto, 2, 0) OVER (ORDER BY fecha)
-- 2 filas atrás, 0 si no existe

-- LEAD: valor de la fila SIGUIENTE
SELECT
    fecha,
    monto,
    LEAD(monto) OVER (ORDER BY fecha) AS proximo
FROM ventas;

-- % crecimiento vs mes anterior
SELECT
    mes,
    ventas,
    ROUND((ventas - LAG(ventas) OVER (ORDER BY mes))
         * 100.0
         / LAG(ventas) OVER (ORDER BY mes), 2) AS pct_cambio
FROM ventas_mensual;
NTILE · FIRST_VALUE · LAST_VALUEintermedio
-- NTILE: divide en N grupos iguales (cuartiles, deciles)
SELECT
    nombre,
    salario,
    NTILE(4) OVER (ORDER BY salario) AS cuartil
FROM empleados;
-- cuartil 1 = más bajos, 4 = más altos

-- FIRST_VALUE: primer valor de la ventana
SELECT
    nombre,
    departamento,
    salario,
    FIRST_VALUE(nombre) OVER (
        PARTITION BY departamento
        ORDER BY salario DESC
    ) AS mejor_pagado_depto
FROM empleados;

-- Truco: top 1 por grupo con ROW_NUMBER en CTE
WITH ranked AS (
    SELECT *,
        ROW_NUMBER() OVER (
            PARTITION BY depto_id
            ORDER BY salario DESC
        ) AS rn
    FROM empleados
)
SELECT * FROM ranked WHERE rn = 1;

    email,

DDL & DML — Crear, Insertar, Actualizar, Eliminar

🏗️
DDL (Data Definition Language): CREATE, ALTER, DROP — define la estructura. DML (Data Manipulation Language): INSERT, UPDATE, DELETE — manipula los datos.
CREATE TABLEbásico
CREATE TABLE empleados (
    id          INT           PRIMARY KEY AUTO_INCREMENT,
    nombre      VARCHAR(100) NOT NULL,
    apellido    VARCHAR(100) NOT NULL,
    email       VARCHAR(200) UNIQUE,
    salario     DECIMAL(10,2) DEFAULT 0,
    depto_id    INT,
    activo      TINYINT      DEFAULT 1,
    fecha_alta  DATE         DEFAULT (CURRENT_DATE),
    FOREIGN KEY (depto_id) REFERENCES departamentos(id)
);

-- Tipos de dato más comunes
-- INT, BIGINT, SMALLINT  → enteros
-- DECIMAL(p,s), FLOAT    → decimales
-- VARCHAR(n), TEXT       → texto
-- DATE, DATETIME, TIME   → fechas
-- BOOLEAN / TINYINT      → verdadero/falso

-- Eliminar columna

¿Qué herramienta usar? — Árbol de Decisión

💡
Esta es la pregunta que todo JR se hace: ¿cuándo uso una subquery, cuándo una CTE y cuándo una window function? Este árbol te da la respuesta según el problema que necesitas resolver.
Árbol de decisión completoref esencial
¿Necesitas FILTRAR basándote en datos de otra tabla?
  ├── SÍ, y la condición es una lista de IDs           → Subquery con IN / NOT IN
  │        Ej: tracks de artistas que... / álbumes que NO tienen...
  │
  └── SÍ, y la condición es un solo valor (AVG, MAX...)   → Subquery escalar en WHERE
           Ej: empleados que ganan más que el promedio

¿Necesitas un CÁLCULO por grupo sin perder el detalle de cada fila?
  └── SÍ  → Window Function con PARTITION BY
           Ej: salario promedio del depto al lado de cada empleado

¿Necesitas NUMERAR, RANKEAR o ver la fila ANTERIOR/SIGUIENTE?
  └── SÍ  → Window Function (ROW_NUMBER / RANK / LAG / LEAD)
           Ej: top 3 vendedores por región

¿Necesitas PREPARAR datos primero (COUNT, AVG) y luego FILTRAR sobre ellos?
  └── SÍ  → CTE
           Ej: departamentos con más de 5 empleados
           Nota: WHERE COUNT(*) > 5 no funciona → WHERE va antes que COUNT

¿La query tiene subqueries anidadas ilegibles o lógica que se repite?
  └── SÍ  → CTE (hace el código legible y reutilizable)

Subqueries — Regla de los Saltos & NOT IN

🧩
Regla de los saltos: pon un dedo en la tabla que VES y otro en la que FILTRA. El número de saltos = número de tablas que necesitas en la subquery. Léelas siempre de adentro hacia afuera.
Subquery escalar — 1 valor calculadointermedio
-- Cuando filtras con AVG, MAX, MIN, COUNT
-- La subquery devuelve exactamente 1 valor

-- Empleados que ganan más que el promedio
SELECT nombre, salario
FROM empleados
WHERE salario > (
    SELECT AVG(salario) FROM empleados
);

-- El empleado con el salario más alto
SELECT nombre, salario
FROM empleados
WHERE salario = (
    SELECT MAX(salario) FROM empleados
);

-- También en SELECT (escalar por cada fila)
SELECT nombre, salario,
    (SELECT AVG(salario) FROM empleados) AS promedio_global
FROM empleados;
-- Regla: devuelve exactamente 1 valor
-- Si devuelve más → SQL lanza error
Subquery con IN — 1 saltointermedio
-- Tabla que VES y tabla que FILTRA
-- están separadas por 1 relación directa

-- Clientes que hicieron pedidos en 2024
SELECT nombre
FROM clientes
WHERE id IN (
    SELECT cliente_id
    FROM pedidos
    WHERE YEAR(fecha) = 2024
);

-- Empleados del depto de 'Ventas'
SELECT nombre
FROM empleados
WHERE depto_id IN (
    SELECT id
    FROM departamentos
    WHERE nombre = 'Ventas'
);
-- Regla de oro: la subquery devuelve el ID
-- que usa la tabla exterior como puente
Subquery con IN — 2 saltosintermedio
-- 2 saltos en el mapa → anidar 2 subqueries
-- Léelas SIEMPRE de adentro hacia afuera

-- Productos vendidos por clientes de 'CDMX'
-- Producto → Pedido → Cliente
SELECT nombre
FROM productos
WHERE id IN (
    SELECT producto_id
    FROM pedidos
    WHERE cliente_id IN (
        SELECT id
        FROM clientes
        WHERE ciudad = 'CDMX'
    )
);

-- Lectura de adentro hacia afuera:
-- 1. clientes devuelve id
-- 2. pedidos recibe cliente_id → devuelve producto_id
-- 3. productos recibe id → muestra nombre
NOT IN — los que no aparecenintermedio
-- Clientes que NUNCA hicieron un pedido
SELECT nombre
FROM clientes
WHERE id NOT IN (
    SELECT cliente_id FROM pedidos
);

-- ⚠️ PELIGRO: si la subquery devuelve algún NULL
-- NOT IN retorna vacío — siempre filtra NULLs
WHERE id NOT IN (
    SELECT cliente_id FROM pedidos
    WHERE cliente_id IS NOT NULL  -- ✅ seguro
)

-- Alternativa más segura: NOT EXISTS
SELECT nombre FROM clientes c
WHERE NOT EXISTS (
    SELECT 1 FROM pedidos p
    WHERE p.cliente_id = c.id
);
-- NOT EXISTS es inmune a NULLs ✅
Las 4 preguntas para resolver cualquier subquerymétodo
1.
¿Qué quiero VER? → tabla del FROM exterior
2.
¿Qué FILTRA? → tabla de la subquery más interna
3.
¿Cuántos SALTOS? → saltos entre tablas = subqueries anidadas
4.
¿IN o NOT IN? → ¿quiero los que SÍ están o los que NO están en la lista?

Window Functions — GROUP BY vs PARTITION BY & ORDER BY en OVER

GROUP BY vs PARTITION BY — la diferencia clavecrítico
GROUP BY ❌ colapsa filas
SELECT depto_id,
    AVG(salario) AS prom
FROM empleados
GROUP BY depto_id;
-- 1 fila por depto
-- Pierdes el nombre del empleado
PARTITION BY ✅ mantiene filas
SELECT nombre, depto_id,
    AVG(salario) OVER (
        PARTITION BY depto_id
    ) AS prom_depto
FROM empleados;
-- 1 fila por empleado
-- + promedio del depto en cada fila ✅
💡 Regla: ¿necesitas el resumen y el detalle en la misma fila? → PARTITION BY. ¿Solo el resumen? → GROUP BY.
ORDER BY dentro vs fuera del OVERcrítico
-- ORDER BY FUERA del OVER
-- → solo ordena lo que ves en pantalla
SELECT nombre, salario
FROM empleados
ORDER BY salario DESC;  -- ordena el resultado final

-- ORDER BY DENTRO del OVER
-- → controla cómo se acumula el cálculo fila a fila
SELECT nombre, salario,
    SUM(salario) OVER (
        PARTITION BY depto_id
        ORDER BY fecha_ingreso  -- ← acumula en este orden
    ) AS salario_acumulado
FROM empleados;
-- Sin ORDER BY en OVER → total fijo del grupo en cada fila
-- Con ORDER BY en OVER → suma crece fila a fila ✅

-- Ambos juntos (muy común en práctica)
SELECT nombre, fecha_ingreso, salario,
    SUM(salario) OVER (
        PARTITION BY depto_id
        ORDER BY fecha_ingreso
    ) AS acumulado
FROM empleados
ORDER BY depto_id, fecha_ingreso;  -- ordena el resultado
ROW_NUMBER vs RANK vs DENSE_RANKintermedio
-- ROW_NUMBER: siempre único (1,2,3,4...)
SELECT nombre,
    ROW_NUMBER() OVER (
        PARTITION BY depto_id
        ORDER BY salario DESC
    ) AS num_fila
FROM empleados;

-- RANK: empates repiten, salta el siguiente (1,2,2,4)
SELECT nombre, salario,
    RANK() OVER (
        PARTITION BY depto_id
        ORDER BY salario DESC
    ) AS ranking
FROM empleados;

-- DENSE_RANK: empates repiten, NO salta (1,2,2,3)
SELECT nombre, salario,
    DENSE_RANK() OVER (
        PARTITION BY depto_id
        ORDER BY salario DESC
    ) AS dense_rank
FROM empleados;

-- Truco: top 1 por grupo sin LIMIT
WITH ranked AS (
    SELECT *,
        ROW_NUMBER() OVER (
            PARTITION BY depto_id
            ORDER BY salario DESC
        ) AS rn
    FROM empleados
)
SELECT * FROM ranked WHERE rn = 1;
Guía ¿cuándo usar cada función de ventana?ref
Si el enunciado dice...UsaORDER BY en OVER
"total / promedio del grupo junto a cada fila"SUM / AVG / COUNT OVER (PARTITION BY)No necesario
"acumulado / progresivo / corriente"SUM / COUNT OVER (PARTITION BY + ORDER BY)Sí, obligatorio
"el primero / único de cada grupo"ROW_NUMBER = 1Sí, obligatorio
"top N / rankear / puede haber empates"RANK o DENSE_RANKSí, obligatorio
"comparar con la fila anterior"LAG(col)Sí, obligatorio
"comparar con la fila siguiente"LEAD(col)Sí, obligatorio

CTEs — Por qué existen & Regla de Conteo

✅
Las CTEs nacen para resolver un problema concreto: no puedes filtrar directamente sobre el resultado de una agregación. WHERE COUNT(*) > 10 nunca funciona porque WHERE se ejecuta ANTES que COUNT. La CTE resuelve esto: primero calculas, luego filtras.
El problema que resuelven las CTEscrítico
-- ❌ Esto NO funciona — WHERE va antes que COUNT
SELECT depto_id, COUNT(*) AS total
FROM empleados
GROUP BY depto_id
WHERE total > 5;  -- ❌ SQL no conoce "total" aquí

-- ✅ Con HAVING (solo para condiciones de agregación)
SELECT depto_id, COUNT(*) AS total
FROM empleados
GROUP BY depto_id
HAVING COUNT(*) > 5;  -- ✅ funciona

-- ✅ Con CTE (cuando necesitas hacer más cosas después)
WITH conteo AS (
    SELECT depto_id, COUNT(*) AS total
    FROM empleados
    GROUP BY depto_id
)
SELECT d.nombre, c.total
FROM conteo AS c
JOIN departamentos AS d ON c.depto_id = d.id
WHERE c.total > 5;  -- ✅ y además le haces JOIN
AVG sobre COUNT — imposible sin CTEintermedio
-- ❌ No puedes anidar funciones de agregación
SELECT AVG(COUNT(*)) FROM empleados;  -- ❌ error

-- ✅ CTE: primero COUNT, luego AVG afuera
WITH conteo AS (
    SELECT depto_id, COUNT(*) AS total
    FROM empleados
    GROUP BY depto_id
)
SELECT AVG(total) AS promedio_por_depto
FROM conteo;

-- Clientes con más de 3 pedidos
WITH frecuencia AS (
    SELECT cliente_id, COUNT(*) AS num_pedidos
    FROM pedidos
    GROUP BY cliente_id
)
SELECT c.nombre, f.num_pedidos
FROM frecuencia AS f
JOIN clientes AS c ON f.cliente_id = c.id
WHERE f.num_pedidos > 3
ORDER BY f.num_pedidos DESC;
Regla de conteo — la más importantemétodo
Fórmula: ¿Cuántos X tiene cada Y?
FROM → qué CUENTAS
FROM X
la entidad que quieres contar
GROUP BY → por quién
GROUP BY Y_id
el agrupador
¿Cuántos PEDIDOS tiene cada CLIENTE?
FROM pedidos · GROUP BY cliente_id

¿Cuántos EMPLEADOS tiene cada DEPTO?
FROM empleados · GROUP BY depto_id

¿Cuántos PRODUCTOS tiene cada CATEGORÍA?
FROM productos · GROUP BY categoria_id

WHERE vs ON & Manejo Completo de NULLs

⚠️
Dos confusiones muy frecuentes: poner filtros en WHERE cuando deben ir en ON (destruye el LEFT JOIN silenciosamente), y comparar NULLs con = en lugar de IS NULL (nunca devuelve filas). El código corre sin error pero los resultados son incorrectos.
WHERE en LEFT JOIN — el error silenciosocrítico
-- ❌ El WHERE convierte LEFT JOIN en INNER JOIN
-- Los clientes sin órdenes desaparecen
SELECT c.nombre, o.monto
FROM clientes AS c
LEFT JOIN ordenes AS o ON c.id = o.cliente_id
WHERE o.monto > 100;  -- ❌ mata el LEFT JOIN

-- ✅ Condición extra en el ON — mantiene todos
SELECT c.nombre, o.monto
FROM clientes AS c
LEFT JOIN ordenes AS o
    ON  c.id = o.cliente_id
    AND o.monto > 100;
-- Clientes sin match → monto = NULL ✅

-- Regla de oro con LEFT JOIN:
-- ON  → condición de unión + filtros del lado derecho
-- WHERE → filtros del resultado FINAL (no del lado derecho)
NULLs — todo lo que necesitas sabercrítico
-- ❌ NUNCA usa = con NULL — siempre retorna vacío
WHERE col = NULL   -- ❌ NULL = NULL → UNKNOWN, no TRUE
WHERE col != NULL  -- ❌ también vacío

-- ✅ Siempre IS NULL / IS NOT NULL
WHERE col IS NULL      -- ✅ filas sin valor
WHERE col IS NOT NULL  -- ✅ filas con valor

-- COALESCE: devuelve el primero no NULL de la lista
SELECT nombre,
    COALESCE(telefono, email, 'Sin contacto') AS contacto
FROM clientes;

-- COALESCE en LEFT JOIN: 0 en vez de NULL
SELECT c.nombre,
    COALESCE(SUM(o.monto), 0) AS total_gastado
FROM clientes AS c
LEFT JOIN ordenes AS o ON c.id = o.cliente_id
GROUP BY c.id, c.nombre;

-- NULLIF: devuelve NULL si dos valores son iguales
-- Evita división por cero
SELECT total / NULLIF(cantidad, 0) FROM ventas;

Errores Comunes — Los más frecuentes en JRs

Errores de filtrado y NULLscríticos
-- ❌ ERROR 1: Comparar NULL con =
WHERE col = NULL    -- ❌ siempre vacío
WHERE col IS NULL   -- ✅

-- ❌ ERROR 2: Usar alias del SELECT en WHERE
SELECT salario * 12 AS anual FROM emp
WHERE anual > 50000;  -- ❌ WHERE va antes que SELECT
WHERE salario * 12 > 50000;  -- ✅ repite la expresión

-- ❌ ERROR 3: ORDER BY dentro de subquery
WHERE id IN (
    SELECT id FROM tabla
    ORDER BY fecha  -- ❌ sin efecto y puede dar error
)
-- ✅ ORDER BY solo en la query exterior

-- ❌ ERROR 4: NOT IN con NULLs en la subquery
WHERE id NOT IN (SELECT manager_id FROM emp)
-- Si hay NULLs en manager_id → resultado vacío ❌
-- ✅ Filtra NULLs en la subquery
WHERE id NOT IN (
    SELECT manager_id FROM emp
    WHERE manager_id IS NOT NULL  -- ✅
)
Errores de agrupación y JOINscríticos
-- ❌ ERROR 5: Columna no agregada fuera del GROUP BY
SELECT nombre, SUM(ventas) FROM emp;  -- ❌
SELECT nombre, SUM(ventas) FROM emp
GROUP BY nombre;  -- ✅

-- ❌ ERROR 6: HAVING para filtros de fila individual
SELECT depto, SUM(sal) FROM emp
GROUP BY depto
HAVING activo = 1;  -- ❌ activo es de fila, no de grupo
-- ✅ activo va antes del GROUP BY
WHERE activo = 1
GROUP BY depto
HAVING SUM(sal) > 50000;  -- ✅ HAVING para la agregación

-- ❌ ERROR 7: JOIN sin ON (producto cartesiano)
SELECT * FROM clientes
JOIN ordenes;  -- ❌ filas × filas = desastre
JOIN ordenes ON clientes.id = ordenes.cliente_id;  -- ✅

-- ❌ ERROR 8: Confundir COUNT(*) vs COUNT(col)
COUNT(*)              -- todas las filas (incluye NULLs)
COUNT(col)            -- solo valores no NULL
COUNT(DISTINCT col)   -- valores únicos no NULL

Ejercicios Reales — Chinook DB & Palabras Clave

🗺️
Mapa Chinook: Artist ──ArtistId──> Album ──AlbumId──> Track. Regla de saltos: pon un dedo en la tabla que VES y otro en la que FILTRA. Cada salto = un IN anidado. Nunca Artist ↔ Track directamente — siempre hay que pasar por Album.
🔑 Palabras clave que te dicen qué herramienta usar ref esencial
→ Subquery con IN
  • 🗣 "que pertenecen a..."
  • 🗣 "tracks del artista..."
  • 🗣 "álbumes que tienen tracks..."
  • 🗣 "de artistas que..."
  • 🗣 "clientes que compraron..."
  • → hay que saltar de una tabla a otra
→ Subquery con NOT IN
  • 🗣 "que NO tienen..."
  • 🗣 "sin ningún..."
  • 🗣 "que nunca han..."
  • 🗣 "artistas sin álbum"
  • 🗣 "que no aparecen en..."
  • → excluir los que sí están en otra tabla
→ Subquery escalar
  • 🗣 "más que el promedio"
  • 🗣 "igual al máximo"
  • 🗣 "mayor que el mínimo"
  • 🗣 "precio promedio de todos"
  • 🗣 "comparar con un solo valor"
  • → la subquery devuelve 1 solo número
→ Window Function
  • 🗣 "junto a / al lado de cada fila"
  • 🗣 "el máximo de su álbum"
  • 🗣 "ranking dentro de su grupo"
  • 🗣 "precio del track anterior"
  • 🗣 "numerar dentro de cada grupo"
  • → cálculo por grupo sin perder el detalle
→ CTE
  • 🗣 "que tienen más de N tracks"
  • 🗣 "con más de X álbumes"
  • 🗣 "cuál es el promedio de..."
  • 🗣 "cuántos X tiene cada Y"
  • 🗣 "filtrar sobre un conteo"
  • → necesitas COUNT/AVG y luego filtrar sobre él
→ JOIN
  • 🗣 "mostrar junto con su..."
  • 🗣 "nombre del artista del track"
  • 🗣 "con la información de..."
  • 🗣 "combinar datos de dos tablas"
  • 🗣 "que incluya columnas de..."
  • → necesitas columnas de más de una tabla
── SUBQUERIES ──────────────────────────────
Tracks del artista 'AC/DC' — 2 saltos IN anidado 
-- Pregunta: ¿Cuáles son los nombres de los tracks
-- que pertenecen a álbumes del artista 'AC/DC'?
-- Ruta: Track → Album → Artist  (2 saltos)

SELECT Name
FROM Track
WHERE AlbumId IN (
    SELECT AlbumId
    FROM Album
    WHERE ArtistId IN (
        SELECT ArtistId
        FROM Artist
        WHERE Name LIKE 'AC/DC'
    )
);

-- Lectura de adentro hacia afuera:
-- 1. Artist devuelve ArtistId
-- 2. Album recibe ArtistId → devuelve AlbumId
-- 3. Track recibe AlbumId → muestra Name
Artistas SIN ningún álbum — NOT IN NOT IN 
-- Pregunta: ¿Cuáles son los artistas que NO tienen
-- ningún álbum registrado?
-- Señal: "NO tienen" → NOT IN
-- Ruta: Artist ───── Album  (1 salto)

SELECT Name
FROM Artist
WHERE ArtistId NOT IN (
    SELECT ArtistId
    FROM Album
);

-- El ArtistId que aparece en Artist
-- pero NO aparece en Album → sin álbum
Álbumes con tracks > 5 min — IN 1 salto IN 
-- Pregunta: ¿Cuáles son los títulos de los álbumes
-- que tienen algún track que dura más de 5 minutos?
-- Señal: "álbumes que tienen tracks que..."
-- Ruta: Album ───── Track  (1 salto)

SELECT Title
FROM Album
WHERE AlbumId IN (
    SELECT AlbumId
    FROM Track
    WHERE Milliseconds > 300000
);

-- 5 minutos = 5 × 60 × 1000 = 300,000 ms
Tracks sobre el precio promedio — escalar escalar 
-- Pregunta: ¿Cuáles son los tracks que cuestan
-- más que el precio promedio de todos los tracks?
-- Señal: "más que el promedio" → subquery escalar
-- La subquery devuelve UN solo número

SELECT TrackId, Name
FROM Track
WHERE UnitPrice > (
    SELECT AVG(UnitPrice)
    FROM Track
);

-- AVG() devuelve 1 valor → usar >, <, =, >=, <=
-- Si devolviera varios → usarías IN
── WINDOW FUNCTIONS ─────────────────────────
Precio máximo del álbum al lado de cada track PARTITION BY 
-- Pregunta: Muestra cada track con su precio y el
-- precio máximo de su álbum
-- Señal: "al lado de cada track" → Window Function
-- No GROUP BY porque necesitas VER cada track

SELECT
    TrackId,
    UnitPrice,
    MAX(UnitPrice) OVER (
        PARTITION BY AlbumId
    ) AS maximo_precio
FROM Track;

-- PARTITION BY AlbumId → calcula el MAX
-- dentro de cada álbum por separado
-- Cada fila conserva su propio UnitPrice ✅
Ranking de precio dentro de cada álbum RANK 
-- Pregunta: Muestra cada track con su nombre y un
-- ranking de precio dentro de su álbum
-- (del más caro al más barato)
-- Señal: "ranking dentro de su..." → RANK + PARTITION BY

SELECT
    TrackId,
    Name,
    RANK() OVER (
        PARTITION BY AlbumId
        ORDER BY UnitPrice DESC
    ) AS ranking_precio
FROM Track;

-- PARTITION BY AlbumId → reinicia el ranking en cada álbum
-- ORDER BY UnitPrice DESC → del más caro al más barato
-- Si dos tracks tienen el mismo precio → mismo rank
Precio del track anterior en el mismo álbum LAG 
-- Pregunta: Muestra cada track con el precio del
-- track anterior dentro de su álbum
-- Señal: "precio del track ANTERIOR" → LAG

SELECT
    TrackId,
    UnitPrice,
    LAG(UnitPrice) OVER (
        PARTITION BY AlbumId
        ORDER BY TrackId
    ) AS precio_anterior
FROM Track;

-- LAG mira la fila anterior dentro del grupo
-- El primer track de cada álbum → precio_anterior = NULL
-- ORDER BY TrackId → define cuál es "anterior"
── CTEs ─────────────────────────────────────
Álbumes con más de 10 tracks CTE + COUNT 
-- Pregunta: ¿Cuáles son los álbumes que tienen
-- más de 10 tracks?
-- Señal: "más de N tracks" → CTE
-- No puedes: WHERE COUNT(*) > 10 ❌

WITH conteo AS (
    SELECT AlbumId, COUNT(*) AS total_tracks
    FROM Track
    GROUP BY AlbumId
)
SELECT AlbumId
FROM conteo
WHERE total_tracks > 10;

-- Regla de conteo:
-- ¿Cuántos TRACKS tiene cada ÁLBUM?
-- FROM Track · GROUP BY AlbumId ✅
Artistas con más de 5 álbumes CTE + COUNT 
-- Pregunta: ¿Cuáles son los artistas que tienen
-- más de 5 álbumes?
-- Señal: "más de N álbumes" → CTE

WITH conteo AS (
    SELECT ArtistId, COUNT(*) AS total_albums
    FROM Album
    GROUP BY ArtistId
)
SELECT ArtistId
FROM conteo
WHERE total_albums > 5;

-- Regla de conteo:
-- ¿Cuántos ÁLBUMES tiene cada ARTISTA?
-- FROM Album · GROUP BY ArtistId ✅
Promedio de tracks por álbum — AVG sobre COUNT CTE + AVG(COUNT) 
-- Pregunta: ¿Cuál es el promedio de tracks por álbum?
-- Señal: "promedio de..." que requiere contar primero
-- No puedes: SELECT AVG(COUNT(*)) ❌ directo

WITH conteo AS (
    SELECT AlbumId, COUNT(*) AS total_tracks
    FROM Track
    GROUP BY AlbumId
)
SELECT AVG(total_tracks) AS promedio
FROM conteo;

-- Paso 1 (CTE): contar tracks por álbum
-- Paso 2 (query): promediar esos conteos
-- AVG(COUNT(*)) ❌ / CTE → luego AVG ✅
🧠 Patrón mental para resolver cualquier ejercicio método 
Paso 1: Lee la pregunta y detecta la palabra clave
  "que pertenecen a / del artista / que tienen..."   → IN
  "que NO tienen / sin ningún / que nunca..."         → NOT IN
  "más que el promedio / igual al máximo..."          → subquery escalar
  "al lado de cada fila / máximo de su grupo..."      → PARTITION BY
  "ranking / numerar / anterior / siguiente..."       → RANK / ROW_NUMBER / LAG / LEAD
  "que tienen más de N / promedio de conteo..."       → CTE

Paso 2: Si usas IN/NOT IN → cuenta los saltos en el mapa
  Track ──AlbumId──> Album ──ArtistId──> Artist
  1 salto  = 1 subquery · 2 saltos = 2 subqueries anidadas
  Lectura: de ADENTRO hacia AFUERA

Paso 3: Si usas CTE → aplica la Regla de Conteo
  ¿Cuántos X tiene cada Y? → FROM X · GROUP BY Y_id
  ¿Cuántos TRACKS tiene cada ÁLBUM? → FROM Track · GROUP BY AlbumId
